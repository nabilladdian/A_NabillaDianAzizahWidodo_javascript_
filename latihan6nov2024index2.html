function divideNumbers() { const num1 = parseFloat(document.getElementById("num1").value); const num2 = parseFloat(document.getElementById("num2").value); if (isNaN(num1) || isNaN(num2)) { displayError("Please enter valid numbers"); } else if (num2 === 0) { displayError("Error: Division by zero is not allowed"); } else { const result = num1 / num2; displayResult(`Result: ${result}`); } } function displayError(message) { document.getElementById("result").innerHTML = message; document.getElementById("result").style.color = "red"; } function displayResult(message) { document.getElementById("result").innerHTML = message; document.getElementById("result").style.color = "green"; }
